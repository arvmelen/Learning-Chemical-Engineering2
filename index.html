# 📘 Title: First-Order Reaction Kinetics
# Description: Models concentration decay over time for a first-order reaction A → products

# 🔹 Step 1: Import Libraries
import numpy as np
import matplotlib.pyplot as plt

# 🔹 Step 2: Define Parameters
k = 0.25  # rate constant (1/s)
CA0 = 1.0  # initial concentration (mol/L)
t = np.linspace(0, 20, 100)  # time array from 0 to 20 seconds

# 🔹 Step 3: Model Equation
CA = CA0 * np.exp(-k * t)  # first-order decay

# 🔹 Step 4: Plot Results
plt.figure(figsize=(8,5))
plt.plot(t, CA, label='CA(t)', color='blue')
plt.xlabel('Time (s)')
plt.ylabel('Concentration CA (mol/L)')
plt.title('First-Order Reaction: Concentration vs Time')
plt.grid(True)
plt.legend()
plt.show()

# Solving and plotting second-order kinetic ODE using solve_ivp

from scipy.integrate import solve_ivp
import numpy as np
import matplotlib.pyplot as plt

# Parameters
k = 0.1  # rate constant (1/(mol·s))
CA0 = 10.0  # initial concentration (mol/L)

# Define the ODE
def dCA_dt(t, CA):
    return [-k * CA[0]**2]

# Time span
t_span = (0, 100)
t_eval = np.linspace(t_span[0], t_span[1], 500)

# Solve the ODE
sol = solve_ivp(dCA_dt, t_span, [CA0], t_eval=t_eval)

# Plot
plt.style.use('seaborn-v0_8')
plt.figure(figsize=(8,5))
plt.plot(sol.t, sol.y[0], label='CA(t)', color='blue')
plt.xlabel('Time (s)')
plt.ylabel('Concentration CA (mol/L)')
plt.title('Second-Order Reaction: CA vs Time')
plt.grid(True)
plt.legend()
plt.tight_layout()

# Solving reversible enzyme kinetics and plotting substrate/product vs time with equilibrium lines

import numpy as np
import matplotlib.pyplot as plt
from scipy.integrate import solve_ivp

# Parameters
vs = 2.0
vp = 1.5
Ks = 5.0
Kp = 10.0
S0 = 100.0
P0 = 0.0

# Rate equation: ds/dt
def rate_equation(t, y):
    s, p = y
    numerator = (vs / Ks) * s - (vp / Kp) * p
    denominator = 1 + s / Ks + p / Kp
    dsdt = numerator / denominator
    dpdt = -dsdt  # Since s + p = constant
    return [dsdt, dpdt]

# Time span and initial conditions
t_span = (0, 50)
y0 = [S0, P0]
t_eval = np.linspace(t_span[0], t_span[1], 500)

# Solve the system
solution = solve_ivp(rate_equation, t_span, y0, t_eval=t_eval)

# Analytical equilibrium concentrations
# (vs/Ks) * seq = (vp/Kp) * (S0 - seq)
# Solve for seq
a = vs / Ks
b = vp / Kp
seq = (b * S0) / (a + b)
peq = S0 - seq

# Plotting
plt.style.use('seaborn-v0_8')
fig, ax = plt.subplots(2, 1, figsize=(10, 8), sharex=True)

# Substrate plot
ax[0].plot(solution.t, solution.y[0], label='[Substrate] s(t)', color='blue')
ax[0].axhline(seq, color='blue', linestyle='--', label=f'Equilibrium s = {seq:.2f}')
ax[0].set_ylabel('Substrate Concentration')
ax[0].legend()
ax[0].grid(True)

# Product plot
ax[1].plot(solution.t, solution.y[1], label='[Product] p(t)', color='green')
ax[1].axhline(peq, color='green', linestyle='--', label=f'Equilibrium p = {peq:.2f}')
ax[1].set_xlabel('Time')
ax[1].set_ylabel('Product Concentration')
ax[1].legend()
ax[1].grid(True)

# Save plot

plt.savefig("enzyme_reaction_equilibrium.png")



plt.tight_layout()

print(f"Equilibrium substrate concentration: {seq:.4f}")
print(f"Equilibrium product concentration: {peq:.4f}")

# Solving reversible enzyme reaction using solve_ivp and plotting s(t) and p(t) with equilibrium lines

import numpy as np
import matplotlib.pyplot as plt
from scipy.integrate import solve_ivp

# Parameters
vs = 2.0
vp = 1.5
Ks = 5.0
Kp = 10.0
S0 = 100.0
P0 = 0.0

# Differential equation
def enzyme_reaction(t, y):
    s, p = y
    numerator = (vs / Ks) * s - (vp / Kp) * p
    denominator = 1 + s / Ks + p / Kp
    rate = numerator / denominator
    dsdt = -rate
    dpdt = rate
    return [dsdt, dpdt]

# Time span
t_span = (0, 50)
t_eval = np.linspace(t_span[0], t_span[1], 500)

# Initial conditions
y0 = [S0, P0]

# Solve ODE
sol = solve_ivp(enzyme_reaction, t_span, y0, t_eval=t_eval)

# Analytical equilibrium concentrations
# (vs/Ks) * seq = (vp/Kp) * (S0 - seq)
# Solve for seq
a = vs / Ks
b = vp / Kp
seq = (b * S0) / (a + b)
peq = S0 - seq

# Plotting
plt.style.use('seaborn-v0_8')
plt.figure(figsize=(10, 6))
plt.plot(sol.t, sol.y[0], label='Substrate [S]', color='blue')
plt.plot(sol.t, sol.y[1], label='Product [P]', color='green')
plt.axhline(seq, color='blue', linestyle='--', label='Equilibrium [S]')
plt.axhline(peq, color='green', linestyle='--', label='Equilibrium [P]')
plt.xlabel('Time (s)', fontsize=12)
plt.ylabel('Concentration (mol/L)', fontsize=12)
plt.title('Reversible Enzyme Reaction: Substrate and Product vs Time', fontsize=14)
plt.legend()
plt.grid(True)
plt.tight_layout()
